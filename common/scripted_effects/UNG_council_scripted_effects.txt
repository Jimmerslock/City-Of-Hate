UNG_recalc_council_icon_frames = {
	set_temp_variable = { unioncouncil_temp = unioncouncil_support }
	divide_temp_variable = { unioncouncil_temp = unioncouncil_total }
	multiply_temp_variable = { unioncouncil_temp = 3 }
	if = {
		limit = { check_variable = { unioncouncil_temp > 0 } }
		clamp_temp_variable = {
			var = unioncouncil_temp
			min = 1
			max = 3
		}
	}
	else = {
		set_temp_variable = { unioncouncil_temp = 1 }
	}
	set_variable = { current_council_icon_frame = unioncouncil_temp }
}
add_timed_idea = {
    idea = unioncouncil_industrialfavour_5
    days = 180
}
unioncouncil_industrialfavour_level_up = {
	custom_effect_tooltip = unioncouncil_industrialfavour_added
	if = {
		limit = { has_idea = unioncouncil_industrialfavour_4 }
			remove_ideas = unioncouncil_industrialfavour_4
			add_timed_idea = {
			idea = unioncouncil_industrialfavour_5
			days = 200
		}
	}
	else_if = {
		limit = { has_idea = unioncouncil_industrialfavour_3 }
			remove_ideas = unioncouncil_industrialfavour_3
			add_timed_idea = {
			idea = unioncouncil_industrialfavour_4
			days = 200
			}
	}
	else_if = {
		limit = { has_idea = unioncouncil_industrialfavour_2 }
			remove_ideas = unioncouncil_industrialfavour_2
			add_timed_idea = {
			idea = unioncouncil_industrialfavour_3
			days = 200
			}
	}
	else_if = {
		limit = { has_idea = unioncouncil_industrialfavour_1 }
			remove_ideas = unioncouncil_industrialfavour_1
			add_timed_idea = {
			idea = unioncouncil_industrialfavour_2
			days = 200
			}
	}
	else_if = {
		limit = { NOT = { has_idea = unioncouncil_industrialfavour_1 } }
			add_timed_idea = {
			idea = unioncouncil_industrialfavour_1
			days = 200
			}
	}
}
