country_event = {#Royal Marriage Call to Arms!
	id = alliance.2
	title = alliance.2.t
	desc = alliance.2.d
	picture = GFX_report_event_hungary_refused
	
	is_triggered_only = yes

	immediate = {
		set_country_flag = already_called_to_rm_war
	}

	option = {#To arms!
		name = alliance.2.a
		ai_chance = {
			factor = 7
			modifier = {
				has_war = yes
				factor = 0.2
			}
		}
			
		clr_country_flag = already_called_to_rm_war
		
		every_country = {
			limit = {
				has_offensive_war_with = FROM
			}
			ROOT = {
				add_to_war = {
					targeted_alliance = FROM
					enemy  = PREV
				}
			}
		}
		
		FROM = {
			country_event = {
				id = alliance.3
				days = 4
			}
		}
	}
	
	option = {#Nay
		name = alliance.2.b
		ai_chance = {
			factor = 4
		}
		
		FROM = {
			country_event = {
				id = alliance.4
				days = 4
			}
		}
		add_political_power = -25
		
		hidden_effect = {
			country_event = {
				id = alliance.5
				days = 70
			}
		}		
		custom_effect_tooltip = decline_rm_war_tt
	}
}
